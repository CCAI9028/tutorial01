import matplotlib.pyplot as plt

# Data
hku_rank_data = {
    "Year": [2021, 2022, 2023, 2024, 2025],
    "QS":   [22,   22,   21,   26,   17],
    "THE":  [39,   30,   31,   35,   35],
}

years = hku_rank_data["Year"]
qs = hku_rank_data["QS"]
the = hku_rank_data["THE"]

# Colors chosen to match the sample style
qs_color = 'gold'
the_color = 'green'

plt.figure(figsize=(6.4, 4.8))

# QS line and area
plt.plot(years, qs, color=qs_color, marker='o', linewidth=2, label='QS')
plt.fill_between(years, qs, max(max(qs), max(the)) + 2,
                 color=qs_color, alpha=0.15)

# THE line and area
plt.plot(years, the, color=the_color, marker='o', linewidth=2, label='THE')
plt.fill_between(years, the, max(max(qs), max(the)) + 2,
                 color=the_color, alpha=0.15)

# Axis labels
plt.xlabel("Year")
plt.ylabel("Rankings")

# Invert y-axis so better rankings (smaller numbers) appear higher
y_min = min(min(qs), min(the)) - 2
y_max = max(max(qs), max(the)) + 2
plt.ylim(y_max, y_min)  # inverted

# Optional: tidy y-ticks (every 5 units)
plt.yticks(range(15, 41, 5))

# Legend in upper right with white semi-transparent background
legend = plt.legend(loc='upper right', frameon=True)
legend.get_frame().set_facecolor('white')
legend.get_frame().set_alpha(0.8)

# Remove grid for a clean look
plt.grid(False)

# Add value labels with rounded, semi-transparent backgrounds
for x, y in zip(years, qs):
    plt.annotate(f"{y}",
                 (x, y),
                 textcoords="offset points",
                 xytext=(0, -20),
                 ha='center',
                 bbox=dict(boxstyle="round,pad=0.3",
                           fc=qs_color, alpha=0.3, ec='none'),
                 color='black')

for x, y in zip(years, the):
    plt.annotate(f"{y}",
                 (x, y),
                 textcoords="offset points",
                 xytext=(0, -20),
                 ha='center',
                 bbox=dict(boxstyle="round,pad=0.3",
                           fc=the_color, alpha=0.3, ec='none'),
                 color='black')

plt.tight_layout()
plt.savefig('code_example_prompt.png')

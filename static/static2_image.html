import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Your data
hku_rank_data = {
    "Year": [2021, 2022, 2023, 2024, 2025],
    "QS": [22, 22, 21, 26, 17],
    "THE": [39, 30, 31, 35, 35],
}

# Convert to DataFrame
df = pd.DataFrame(hku_rank_data)

# Plotting QS
plt.figure(figsize=(8, 6))
ax = sns.lineplot(
    x='Year', y='QS',
    color='g',
    marker='o',
    markersize=6,
    data=df,
    label='QS'
)

# Plotting THE
sns.lineplot(
    x='Year', y='THE',
    color='b',
    marker='o',
    markersize=6,
    data=df,
    label='THE'
)

# Invert Y-axis for both lines
ax.invert_yaxis()

# Set X and Y ticks
ax.set(
    xticks=df['Year'],
    yticks=[1, 15, 30, 45]
)

# Legend in upper right
ax.legend(loc='upper right')

# Fill under QS line
plt.fill_between(
    x=df['Year'],
    y1=df['QS'],
    y2=[45] * len(df['QS']),
    alpha=0.1,
    facecolor='g'
)

# Fill under THE line
plt.fill_between(
    x=df['Year'],
    y1=df['THE'],
    y2=[45] * len(df['THE']),
    alpha=0.1,
    facecolor='b'
)

# Remove top and right spines
sns.despine()

# Add annotations for QS
for year, rank in zip(df['Year'], df['QS']):
    plt.text(
        year, rank - 1.5, str(rank), fontsize=10, c='g',
        bbox=dict(
            boxstyle="round",
            ec='g',
            fc='g',
            alpha=0.05
        )
    )

# Add annotations for THE
for year, rank in zip(df['Year'], df['THE']):
    plt.text(
        year, rank - 1.5, str(rank), fontsize=10, c='b',
        bbox=dict(
            boxstyle="round",
            ec='b',
            fc='b',
            alpha=0.05
        )
    )

# Set axis limits
ax.set_ylim(bottom=45, top=5)
ax.set_xlim(left=df['Year'].iloc[0] - 0.1, right=df['Year'].iloc[-1] + 0.1)

# Customize spines and tick parameters
ax.spines["bottom"].set_linewidth(0.5)
ax.spines["left"].set_linewidth(0.5)
ax.tick_params(width=0)

# Show plot
plt.savefig('image_example_prompt.png')

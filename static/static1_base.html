import matplotlib.pyplot as plt
import pandas as pd

# Data
hku_rank_data = {
    "Year": [2021, 2022, 2023, 2024, 2025],
    "QS": [22, 22, 21, 26, 17],
    "THE": [39, 30, 31, 35, 35],
}

# Convert to DataFrame
df = pd.DataFrame(hku_rank_data)

# Plot
plt.figure(figsize=(10, 6))
plt.plot(df['Year'], df['QS'], marker='o', label='QS')
plt.plot(df['Year'], df['THE'], marker='s', label='THE')

# Labels and title
plt.xlabel('Year')
plt.ylabel('Ranking')
plt.title('HKU Rankings Over Years')
plt.gca().invert_yaxis()  # Rankings are better when lower
plt.legend()
plt.grid(True)
plt.tight_layout()

# Show plot
plt.savefig('base_prompt.png')
